Nous pouvons insérer de beaux graphiques, comme celui de la
figure\vref{sin-x*sin-y}.
\begin{figure}[ht]
  \centering
  \capstart
  \begin{tikzpicture}
    \begin{axis}[colorbar]
      \addplot3[surf,domain=0:360]
      {sin(x)*sin(y)};
    \end{axis}
  \end{tikzpicture}
  \caption{Représentation graphique de la fonction $f:(x,y)\mapsto
    \sin x\times\sin y$}
  \label{sin-x*sin-y}
\end{figure}
